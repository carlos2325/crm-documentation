Configuraci贸n de headers para autenticaci贸n JWT con whitelabel.

JavaScript/Node.js

// Configuraci贸n de cliente GraphQL
import { ApolloClient, InMemoryCache, createHttpLink } from '@apollo/client';
import { setContext } from '@apollo/client/link/context';

const httpLink = createHttpLink({
  uri: 'https://api2.eventosorganizador.com/graphql',
});

const authLink = setContext((_, { headers }) => {
  // Obtener token JWT (desde localStorage, cookies, etc.)
  const token = localStorage.getItem('jwt_token');
  const whitelabelId = localStorage.getItem('whitelabel_id');
  
  return {
    headers: {
      ...headers,
      'Authorization': token ? `Bearer ${token}` : "",
      'x-whitelabel-id': whitelabelId || 'default',
      'Content-Type': 'application/json',
    }
  }
});

const client = new ApolloClient({
  link: authLink.concat(httpLink),
  cache: new InMemoryCache()
});

export default client;

// Cliente Apollo con error handling
import { ApolloClient, InMemoryCache, createHttpLink, from } from '@apollo/client';
import { onError } from '@apollo/client/link/error';
import { RetryLink } from '@apollo/client/link/retry';

// Error Link
const errorLink = onError(({ graphQLErrors, networkError, operation, forward }) => {
  if (graphQLErrors) {
    graphQLErrors.forEach(({ message, locations, path }) => {
      console.error(`GraphQL error: ${message}`);
      
      // Errores espec铆ficos
      if (message.includes('Authentication failed')) {
        // Redirigir a login
        window.location.href = '/login';
      }
      
      if (message.includes('Whitelabel not found')) {
        console.error('Configuraci贸n de tenant incorrecta');
      }
    });
  }

  if (networkError) {
    console.error(`Network error: ${networkError}`);
    
    // Reintento autom谩tico para errores de red
    if (networkError.statusCode === 500) {
      return forward(operation);
    }
  }
});

// Retry Link para reconexi贸n autom谩tica
const retryLink = new RetryLink({
  delay: {
    initial: 300,
    max: Infinity,
    jitter: true
  },
  attempts: {
    max: 5,
    retryIf: (error, _operation) => !!error
  }
});

// Cliente con error handling
const client = new ApolloClient({
  link: from([errorLink, retryLink, authLink, httpLink]),
  cache: new InMemoryCache(),
  defaultOptions: {
    watchQuery: {
      errorPolicy: 'all'
    },
    query: {
      errorPolicy: 'all'
    }
  }

// Wrapper para queries con try-catch
async function safeQuery(query, variables = {}) {
  try {
    const { data, errors } = await client.query({
      query,
      variables,
      fetchPolicy: 'network-only'
    });
    
    // Verificar errores de negocio
    if (errors && errors.length > 0) {
      console.warn('GraphQL errors:', errors);
    }
    
    return { data, errors: errors || [] };
    
  } catch (error) {
    console.error('Query failed:', error);
    
    // Errores espec铆ficos de Apollo
    if (error.networkError) {
      return { 
        data: null, 
        errors: [{ message: 'Error de conexi贸n. Verifica tu internet.' }] 
      };
    }
    
    return { 
      data: null, 
      errors: [{ message: error.message }] 
    };
  }
}

// Ejemplo de uso
async function getLeadsWithErrorHandling() {
  const GET_LEADS = gql`
    query GetLeads {
      getCRMLeads(pagination: {limit: 10}) {
        success
        leads {
          id
          name
          email
        }
        errors
      }
    }
  `;
  
  const { data, errors } = await safeQuery(GET_LEADS);
  
  if (data && data.getCRMLeads.success) {
    return data.getCRMLeads.leads;
  } else {
    const allErrors = [
      ...errors,
      ...(data?.getCRMLeads?.errors || [])
    ];
    throw new Error(`Failed to get leads: ${allErrors.join(', ')}`);
  }
}
